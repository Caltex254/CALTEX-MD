   775‚Üí  async handleMessage(sock: WASocket, data: any, sessionId: string): Promise<void> {
   776‚Üí    try {
   777‚Üí      const { messages, type } = data;
   778‚Üí
   779‚Üí      for (const message of messages) {
   780‚Üí        if (!message.message) continue;
   781‚Üí        if (message.key.fromMe) continue;
   782‚Üí
   783‚Üí        const messageId = message.key.id;
   784‚Üí        if (!messageId || this.processedMessages.has(messageId)) continue;
   785‚Üí
   786‚Üí        this.processedMessages.add(messageId);
   787‚Üí        if (this.processedMessages.size > 10000) {
   788‚Üí          const entries = Array.from(this.processedMessages);
   789‚Üí          this.processedMessages = new Set(entries.slice(-5000));
   790‚Üí        }
   791‚Üí
   792‚Üí        const jid = message.key.remoteJid!;
   793‚Üí        const sender = message.key.participant || message.key.remoteJid!;
   794‚Üí
   795‚Üí        if (this.config.blockedJids.includes(sender)) {
   796‚Üí          logger.info({ sender }, 'Message from blocked user, ignoring');
   797‚Üí          continue;
   798‚Üí        }
   799‚Üí
   800‚Üí        // Auto read
   801‚Üí        if (this.config.autoRead) {
   802‚Üí          try {
   803‚Üí            await sock.readMessages([message.key]);
   804‚Üí          } catch (err) {
   805‚Üí            logger.debug({ err }, 'Auto-read failed');
   806‚Üí          }
   807‚Üí        }
   808‚Üí
   809‚Üí        // Auto status view
   810‚Üí        if (jid === 'status@broadcast' && this.config.autoStatusView) {
   811‚Üí          try {
   812‚Üí            await sock.readMessages([message.key]);
   813‚Üí          } catch (err) {
   814‚Üí            logger.debug({ err }, 'Auto status view failed');
   815‚Üí          }
   816‚Üí          continue;
   817‚Üí        }
   818‚Üí
   819‚Üí        const isGroup = jid.endsWith('@g.us');
   820‚Üí        let groupMetadata: GroupMetadata | undefined;
   821‚Üí        let isAdmin = false;
   822‚Üí        let isBotAdmin = false;
   823‚Üí
   824‚Üí        if (isGroup) {
   825‚Üí          groupMetadata = await this.getGroupMetadata(sock, jid);
   826‚Üí          const participant = groupMetadata?.participants?.find((p) => p.id === sender);
   827‚Üí          isAdmin = participant?.admin === 'admin' || participant?.admin === 'superadmin';
   828‚Üí          const botJid = sock.user?.id?.replace(/:.*@/, '@');
   829‚Üí          const botParticipant = groupMetadata?.participants?.find((p) => p.id === botJid || p.id?.split('@')[0] === sock.user?.id?.split(':')[0]);
   830‚Üí          isBotAdmin = botParticipant?.admin === 'admin' || botParticipant?.admin === 'superadmin';
   831‚Üí        }
   832‚Üí
   833‚Üí        const isOwner = this.config.ownerJids.includes(sender) || this.config.ownerJids.includes(sender.split('@')[0]);
   834‚Üí
   835‚Üí        // Anti-features check
   836‚Üí        if (isGroup) {
   837‚Üí          const antiLinkResult = await this.antiFeatures.checkAntiLink(sock, message, groupMetadata);
   838‚Üí          if (antiLinkResult) continue;
   839‚Üí
   840‚Üí          const antiBadwordResult = await this.antiFeatures.checkAntiBadword(sock, message, groupMetadata);
   841‚Üí          if (antiBadwordResult) continue;
   842‚Üí
   843‚Üí          const antiSpamResult = await this.antiFeatures.checkAntiSpam(sock, message, groupMetadata);
   844‚Üí          if (antiSpamResult) continue;
   845‚Üí
   846‚Üí          const antiTagResult = await this.antiFeatures.checkAntiTag(sock, message, groupMetadata);
   847‚Üí          if (antiTagResult) continue;
   848‚Üí        }
   849‚Üí
   850‚Üí        // Capture view-once media
   851‚Üí        if (this.antiFeatures.getConfig().antiViewOnce.enabled) {
   852‚Üí          await this.antiFeatures.handleViewOnceMessage(sock, message);
   853‚Üí        }
   854‚Üí
   855‚Üí        // Auto react
   856‚Üí        if (this.config.autoReact.enabled) {
   857‚Üí          const emojis = this.config.autoReact.emojis;
   858‚Üí          const randomEmoji = emojis[Math.floor(Math.random() * emojis.length)];
   859‚Üí          try {
   860‚Üí            await this.mediaHandler.sendReaction(sock, jid, messageId!, randomEmoji);
   861‚Üí          } catch (err) {
   862‚Üí            logger.debug({ err }, 'Auto-react failed');
   863‚Üí          }
   864‚Üí        }
   865‚Üí
   866‚Üí        // Extract text
   867‚Üí        const text = this.antiFeatures.extractText(message);
   868‚Üí        if (!text) continue;
   869‚Üí
   870‚Üí        // Auto-reply check
