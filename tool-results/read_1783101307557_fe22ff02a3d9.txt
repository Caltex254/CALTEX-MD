     1→// ============================================================================
     2→// CALTEX MD WhatsApp Bot - Main Entry Point
     3→// Starts the HTTP server on port 3031 and initializes the WhatsApp connection
     4→// ============================================================================
     5→
     6→import { createServer, IncomingMessage, ServerResponse } from 'http';
     7→import { SessionManager } from './src/session-manager';
     8→import { ConnectionManager, logger } from './src/connection';
     9→import { AntiFeatures } from './src/anti-features';
    10→import { MediaHandler } from './src/media-handler';
    11→import { AIHandler } from './src/ai-handler';
    12→import { GroupManager } from './src/group-manager';
    13→import { MessageHandler } from './src/message-handler';
    14→import { Scheduler } from './src/scheduler';
    15→import { APIClient } from './src/api-client';
    16→
    17→const PORT = 3031;
    18→const DEFAULT_SESSION_ID = 'caltex-md';
    19→
    20→class CaltexBot {
    21→  private sessionManager: SessionManager;
    22→  private connectionManager: ConnectionManager;
    23→  private antiFeatures: AntiFeatures;
    24→  private mediaHandler: MediaHandler;
    25→  private aiHandler: AIHandler;
    26→  private groupManager: GroupManager;
    27→  private messageHandler: MessageHandler;
    28→  private scheduler: Scheduler;
    29→  private apiClient: APIClient;
    30→  private httpServer: ReturnType<typeof createServer>;
    31→  private startTime: number;
    32→  private totalMessagesProcessed = 0;
    33→  private totalCommandsExecuted = 0;
    34→
    35→  constructor() {
    36→    this.startTime = Date.now();
    37→
    38→    logger.info('='.repeat(50));
    39→    logger.info('  CALTEX MD WhatsApp Bot - Starting...');
    40→    logger.info('='.repeat(50));
    41→
    42→    // Initialize all components
    43→    this.sessionManager = new SessionManager();
    44→    this.connectionManager = new ConnectionManager(this.sessionManager);
    45→    this.antiFeatures = new AntiFeatures();
    46→    this.mediaHandler = new MediaHandler();
    47→    this.aiHandler = new AIHandler();
    48→    this.groupManager = new GroupManager();
    49→    this.scheduler = new Scheduler();
    50→    this.apiClient = new APIClient();
    51→
    52→    // Initialize message handler with all dependencies
    53→    this.messageHandler = new MessageHandler(
    54→      this.antiFeatures,
    55→      this.mediaHandler,
    56→      this.aiHandler,
    57→      this.groupManager
    58→    );
    59→
    60→    // Setup event handlers
    61→    this.setupEventHandlers();
    62→
    63→    // Setup HTTP server for health checks and API
    64→    this.httpServer = this.createHTTPServer();
    65→
    66→    // Setup graceful shutdown
    67→    this.setupShutdownHandlers();
    68→  }
    69→
    70→  private setupEventHandlers(): void {
    71→    // Connection events
    72→    this.connectionManager.on('connection.open', (sessionId: string) => {
    73→      logger.info({ sessionId }, 'WhatsApp connection opened');
    74→      this.sessionManager.updateSessionStatus(sessionId, 'active');
    75→      this.apiClient.reportConnectionStatus(sessionId, 'connected');
    76→      this.apiClient.sendLog('info', 'connection', 'WhatsApp connection opened', { sessionId });
    77→    });
    78→
    79→    this.connectionManager.on('connection.close', (statusCode: number, reason: string, sessionId: string) => {
    80→      logger.warn({ sessionId, statusCode, reason }, 'WhatsApp connection closed');
    81→      this.sessionManager.updateSessionStatus(sessionId, 'disconnected');
    82→      this.apiClient.reportConnectionStatus(sessionId, 'disconnected', { statusCode, reason });
    83→      this.apiClient.sendLog('warn', 'connection', 'WhatsApp connection closed', { sessionId, statusCode, reason });
    84→      // Notify owner about disconnection
    85→      this.apiClient.notifyOwner('disconnect', 'Bot Disconnected', `Session ${sessionId} disconnected: ${reason} (code: ${statusCode})`, 'error');
    86→    });
    87→
    88→    this.connectionManager.on('qr.code', (qr: string, sessionId: string) => {
    89→      logger.info({ sessionId }, 'QR code generated - scan with WhatsApp');
    90→      this.apiClient.reportQRCode(sessionId, qr);
    91→    });
    92→
    93→    this.connectionManager.on('pairing.code', (code: string, sessionId: string, phoneNumber: string) => {
    94→      logger.info({ sessionId, phoneNumber, code }, 'Pairing code generated');
    95→      this.apiClient.sendLog('info', 'pairing', `Pairing code generated for ${phoneNumber}: ${code}`, { sessionId, phoneNumber });
    96→    });
    97→
    98→    // Message events
    99→    this.connectionManager.on('messages.upsert', async (data: any, sessionId: string) => {
   100→      try {
   101→        await this.messageHandler.handleMessage(
   102→          this.connectionManager.getSocket(sessionId)!,
   103→          data,
   104→          sessionId
   105→        );
   106→        this.totalMessagesProcessed++;
   107→
   108→        if (this.totalMessagesProcessed % 100 === 0) {
   109→          this.apiClient.updateStats({
   110→            totalMessages: this.totalMessagesProcessed,
   111→          });
   112→        }
   113→      } catch (err) {
   114→        logger.error({ err, sessionId }, 'Error handling messages.upsert');
   115→        this.apiClient.sendLog('error', 'message-handler', 'Error processing messages', { error: String(err) });
   116→        this.apiClient.notifyOwner('plugin_crash', 'Message Handler Error', `Error in message processing: ${String(err)}`, 'error');
   117→      }
   118→    });
   119→
   120→    this.connectionManager.on('messages.delete', async (data: any, sessionId: string) => {
   121→      try {
   122→        if (this.antiFeatures.getConfig().antiDelete.enabled) {
   123→          const sock = this.connectionManager.getSocket(sessionId);
   124→          if (sock && data.keys) {
   125→            for (const key of data.keys) {
   126→              await this.antiFeatures.handleAntiDelete(sock, key.id, key.remoteJid);
   127→            }
   128→          }
   129→        }
   130→      } catch (err) {
   131→        logger.error({ err }, 'Error handling messages.delete');
   132→      }
   133→    });
   134→
   135→    // Group participant events
   136→    this.connectionManager.on('group.participants.update', async (data: any, sessionId: string) => {
   137→      try {
   138→        const sock = this.connectionManager.getSocket(sessionId);
   139→        if (sock) {
   140→          await this.messageHandler.handleGroupParticipantsUpdate(sock, data, sessionId);
   141→        }
   142→      } catch (err) {
   143→        logger.error({ err }, 'Error handling group.participants.update');
   144→      }
   145→    });
   146→
   147→    // Call events
   148→    this.connectionManager.on('call', async (data: any, sessionId: string) => {
   149→      try {
   150→        const sock = this.connectionManager.getSocket(sessionId);
   151→        if (sock) {
   152→          await this.messageHandler.handleCall(sock, data, sessionId);
   153→        }
   154→      } catch (err) {
   155→        logger.error({ err }, 'Error handling call event');
   156→      }
   157→    });
   158→
   159→    // Connection update events
   160→    this.connectionManager.on('connection.update', (update: any, sessionId: string) => {
   161→      if (update.connection === 'connecting') {
   162→        this.apiClient.reportConnectionStatus(sessionId, 'connecting');
   163→      }
   164→    });
   165→
   166→    // Message handler events
   167→    this.messageHandler.on('command:executed', (data: any) => {
   168→      this.totalCommandsExecuted++;
   169→      this.apiClient.reportCommand(data.command, data.sender, data.jid, data.success);
   170→    });
   171→
   172→    this.messageHandler.on('message:processed', (data: any) => {
   173→      this.emit('stats:update', data);
   174→    });
   175→
   176→    // Scheduler events
   177→    this.scheduler.on('message:sent', (msg: any) => {
   178→      this.apiClient.sendLog('info', 'scheduler', 'Scheduled message sent', { messageId: msg.id, jid: msg.jid });
   179→    });
   180→
   181→    this.scheduler.on('message:failed', (msg: any) => {
   182→      this.apiClient.sendLog('error', 'scheduler', 'Scheduled message failed', { messageId: msg.id, error: msg.error });
   183→    });
   184→  }
   185→
   186→  private createHTTPServer(): ReturnType<typeof createServer> {
   187→    const server = createServer(async (req: IncomingMessage, res: ServerResponse) => {
   188→      const url = req.url ?? '/';
   189→      const method = req.method ?? 'GET';
   190→
   191→      // CORS headers
   192→      res.setHeader('Access-Control-Allow-Origin', '*');
   193→      res.setHeader('Access-Control-Allow-Methods', 'GET, POST, PUT, DELETE, OPTIONS');
   194→      res.setHeader('Access-Control-Allow-Headers', 'Content-Type, Authorization');
   195→
   196→      if (method === 'OPTIONS') {
   197→        res.writeHead(204);
   198→        res.end();
   199→        return;
   200→      }
   201→
   202→      try {
   203→        // Parse request body for POST/PUT
   204→        let body: any = null;
   205→        if (method === 'POST' || method === 'PUT') {
   206→          const chunks: Buffer[] = [];
   207→          for await (const chunk of req) {
   208→            chunks.push(Buffer.from(chunk));
   209→          }
   210→          const rawBody = Buffer.concat(chunks).toString();
   211→          if (rawBody) {
   212→            try {
   213→              body = JSON.parse(rawBody);
   214→            } catch {
   215→              body = rawBody;
   216→            }
   217→          }
   218→        }
   219→
   220→        // --- Route Handling ---
   221→
   222→        // Health check
   223→        if (url === '/health' && method === 'GET') {
   224→          const uptime = Date.now() - this.startTime;
   225→          const connections = this.connectionManager.listConnections();
   226→          const connectedSessions = connections.filter((id) => this.connectionManager.isConnected(id));
   227→          const schedulerStats = this.scheduler.getStats();
   228→
   229→          res.writeHead(200, { 'Content-Type': 'application/json' });
   230→          res.end(JSON.stringify({
   231→            status: 'ok',
   232→            service: 'caltex-bot',
   233→            version: '1.0.0',
   234→            uptime,
   235→            uptimeFormatted: this.formatUptime(uptime),
   236→            connections: {
   237→              total: connections.length,
   238→              connected: connectedSessions.length,
   239→              sessions: connections,
   240→            },
   241→            messages: {
   242→              processed: this.totalMessagesProcessed,
   243→              commandsExecuted: this.totalCommandsExecuted,
   244→            },
   245→            scheduler: schedulerStats,
   246→            antiFeatures: {
   247→              deletedMessages: this.antiFeatures.getDeletedMessageCount(),
   248→              viewOnceMessages: this.antiFeatures.getViewOnceMessageCount(),
   249→            },
   250→            ai: {
   251→              activeConversations: this.aiHandler.listActiveConversations().length,
   252→            },
   253→            timestamp: Date.now(),
   254→          }));
   255→          return;
   256→        }
   257→
   258→        // Status
   259→        if (url === '/api/status' && method === 'GET') {
   260→          const sessions = this.sessionManager.getAllSessionInfo();
   261→          res.writeHead(200, { 'Content-Type': 'application/json' });
   262→          res.end(JSON.stringify({ sessions, timestamp: Date.now() }));
   263→          return;
   264→        }
   265→
   266→        // QR status
   267→        if (url === '/api/qr' && method === 'GET') {
   268→          res.writeHead(200, { 'Content-Type': 'application/json' });
   269→          res.end(JSON.stringify({
   270→            message: 'QR code is displayed in the console. Check bot logs.',
   271→            hint: 'Scan with WhatsApp > Linked Devices > Link a device',
   272→          }));
   273→          return;
   274→        }
   275→
   276→        // Pairing code - request a pairing code for phone number linking
   277→        if (url === '/api/pairing-code' && method === 'POST') {
   278→          const phoneNumber = body?.phoneNumber;
   279→          const sessionId = body?.sessionId ?? DEFAULT_SESSION_ID;
   280→          if (!phoneNumber) {
   281→            res.writeHead(400, { 'Content-Type': 'application/json' });
   282→            res.end(JSON.stringify({ success: false, error: 'phoneNumber is required' }));
   283→            return;
   284→          }
   285→          try {
   286→            const pairingCode = await this.connectionManager.requestPairingCode(sessionId, phoneNumber);
   287→            res.writeHead(200, { 'Content-Type': 'application/json' });
   288→            res.end(JSON.stringify({
   289→              success: true,
   290→              data: { pairingCode, phoneNumber, sessionId },
   291→              instructions: [
   292→                '1. Open WhatsApp on your phone',
   293→                '2. Go to Settings > Linked Devices',
   294→                '3. Tap "Link a Device"',
   295→                '4. Enter the pairing code shown above',
   296→              ],
   297→            }));
   298→          } catch (err: any) {
   299→            res.writeHead(500, { 'Content-Type': 'application/json' });
   300→            res.end(JSON.stringify({ success: false, error: err.message }));
   301→          }
   302→          return;
   303→        }
   304→
   305→        // Connect with pairing code
   306→        if (url === '/api/connect-pairing' && method === 'POST') {
   307→          const phoneNumber = body?.phoneNumber;
   308→          const sessionId = body?.sessionId ?? DEFAULT_SESSION_ID;
   309→          if (!phoneNumber) {
   310→            res.writeHead(400, { 'Content-Type': 'application/json' });
   311→            res.end(JSON.stringify({ success: false, error: 'phoneNumber is required' }));
   312→            return;
   313→          }
   314→          try {
   315→            const result = await this.connectionManager.createConnectionWithPairingCode(
   316→              { sessionId, printQR: false },
   317→              phoneNumber
   318→            );
   319→            res.writeHead(200, { 'Content-Type': 'application/json' });
   320→            res.end(JSON.stringify({
   321→              success: true,
   322→              data: { pairingCode: result.pairingCode, phoneNumber, sessionId },
   323→              instructions: [
   324→                '1. Open WhatsApp on your phone',
   325→                '2. Go to Settings > Linked Devices',
   326→                '3. Tap "Link a Device"',
   327→                '4. Enter the pairing code: ' + result.pairingCode,
   328→              ],
   329→            }));
   330→          } catch (err: any) {
   331→            res.writeHead(500, { 'Content-Type': 'application/json' });
   332→            res.end(JSON.stringify({ success: false, error: err.message }));
   333→          }
   334→          return;
   335→        }
   336→
   337→        // Connect session
   338→        if (url === '/api/connect' && method === 'POST') {
   339→          const sessionId = body?.sessionId ?? DEFAULT_SESSION_ID;
   340→          await this.connect(sessionId);
   341→          res.writeHead(200, { 'Content-Type': 'application/json' });
   342→          res.end(JSON.stringify({ success: true, message: `Connecting session: ${sessionId}` }));
   343→          return;
   344→        }
   345→
   346→        // Disconnect session
   347→        if (url === '/api/disconnect' && method === 'POST') {
   348→          const sessionId = body?.sessionId ?? DEFAULT_SESSION_ID;
   349→          await this.connectionManager.disconnect(sessionId);
   350→          res.writeHead(200, { 'Content-Type': 'application/json' });
   351→          res.end(JSON.stringify({ success: true, message: `Disconnected session: ${sessionId}` }));
   352→          return;
   353→        }
   354→
   355→        // List sessions
   356→        if (url === '/api/sessions' && method === 'GET') {
   357→          const sessions = this.sessionManager.listSessions();
   358→          const sessionDetails = sessions.map((id) => ({
   359→            id,
   360→            info: this.sessionManager.getSessionStatus(id),
   361→            connected: this.connectionManager.isConnected(id),
   362→            reconnectAttempts: this.connectionManager.getReconnectAttempts(id),
   363→          }));
   364→          res.writeHead(200, { 'Content-Type': 'application/json' });
   365→          res.end(JSON.stringify({ sessions: sessionDetails }));
   366→          return;
   367→        }
   368→
   369→        // Create session
   370→        if (url === '/api/sessions/create' && method === 'POST') {
   371→          const sessionId = body?.sessionId;
   372→          if (!sessionId) {
   373→            res.writeHead(400, { 'Content-Type': 'application/json' });
   374→            res.end(JSON.stringify({ error: 'sessionId is required' }));
   375→            return;
   376→          }
   377→          const info = await this.sessionManager.createSession(sessionId);
   378→          res.writeHead(200, { 'Content-Type': 'application/json' });
   379→          res.end(JSON.stringify({ success: true, session: info }));
   380→          return;
   381→        }
   382→
   383→        // Delete session
   384→        if (url === '/api/sessions/delete' && method === 'POST') {
   385→          const sessionId = body?.sessionId;
   386→          if (!sessionId) {
   387→            res.writeHead(400, { 'Content-Type': 'application/json' });
   388→            res.end(JSON.stringify({ error: 'sessionId is required' }));
   389→            return;
   390→          }
   391→          await this.connectionManager.disconnect(sessionId);
   392→          const deleted = await this.sessionManager.deleteSession(sessionId);
   393→          res.writeHead(200, { 'Content-Type': 'application/json' });
   394→          res.end(JSON.stringify({ success: deleted }));
   395→          return;
   396→        }
   397→
   398→        // Export session
   399→        if (url === '/api/sessions/export' && method === 'POST') {
   400→          const sessionId = body?.sessionId;
   401→          if (!sessionId) {
   402→            res.writeHead(400, { 'Content-Type': 'application/json' });
   403→            res.end(JSON.stringify({ error: 'sessionId is required' }));
   404→            return;
   405→          }
   406→          const data = await this.sessionManager.exportSession(sessionId);
   407→          res.writeHead(200, { 'Content-Type': 'application/json' });
   408→          res.end(JSON.stringify({ success: !!data, data }));
   409→          return;
   410→        }
   411→
   412→        // Import session
   413→        if (url === '/api/sessions/import' && method === 'POST') {
   414→          const { sessionId, data } = body ?? {};
   415→          if (!sessionId || !data) {
   416→            res.writeHead(400, { 'Content-Type': 'application/json' });
   417→            res.end(JSON.stringify({ error: 'sessionId and data are required' }));
   418→            return;
   419→          }
   420→          const success = await this.sessionManager.importSession(sessionId, data);
   421→          res.writeHead(200, { 'Content-Type': 'application/json' });
   422→          res.end(JSON.stringify({ success }));
   423→          return;
   424→        }
   425→
   426→        // Backup all sessions
   427→        if (url === '/api/backup' && method === 'POST') {
   428→          const backups = await this.sessionManager.backupAllSessions();
   429→          res.writeHead(200, { 'Content-Type': 'application/json' });
   430→          res.end(JSON.stringify({ success: true, backups: backups.length }));
   431→          return;
   432→        }
   433→
   434→        // List commands
   435→        if (url === '/api/commands' && method === 'GET') {
   436→          const commands = this.messageHandler.listCommands();
   437→          res.writeHead(200, { 'Content-Type': 'application/json' });
   438→          res.end(JSON.stringify({ commands }));
   439→          return;
   440→        }
   441→
   442→        // Bot config
   443→        if (url === '/api/config/bot' && method === 'GET') {
   444→          const config = this.messageHandler.getConfig();
   445→          res.writeHead(200, { 'Content-Type': 'application/json' });
   446→          res.end(JSON.stringify({ config }));
   447→          return;
   448→        }
   449→
   450→        if (url === '/api/config/bot' && method === 'PUT') {
   451→          if (!body) {
   452→            res.writeHead(400, { 'Content-Type': 'application/json' });
   453→            res.end(JSON.stringify({ error: 'Request body is required' }));
   454→            return;
   455→          }
   456→          this.messageHandler.updateConfig(body);
   457→          res.writeHead(200, { 'Content-Type': 'application/json' });
   458→          res.end(JSON.stringify({ success: true }));
   459→          return;
   460→        }
   461→
   462→        // Anti-feature config
   463→        if (url === '/api/config/anti' && method === 'GET') {
   464→          const config = this.antiFeatures.getConfig();
   465→          res.writeHead(200, { 'Content-Type': 'application/json' });
   466→          res.end(JSON.stringify({ config }));
   467→          return;
   468→        }
   469→
   470→        if (url === '/api/config/anti' && method === 'PUT') {
   471→          if (!body) {
   472→            res.writeHead(400, { 'Content-Type': 'application/json' });
   473→            res.end(JSON.stringify({ error: 'Request body is required' }));
   474→            return;
   475→          }
   476→          this.antiFeatures.updateConfig(body);
   477→          res.writeHead(200, { 'Content-Type': 'application/json' });
   478→          res.end(JSON.stringify({ success: true }));
   479→          return;
   480→        }
   481→
   482→        // AI config
   483→        if (url === '/api/config/ai' && method === 'GET') {
   484→          const config = this.aiHandler.getConfig();
   485→          const masked = { ...config } as any;
   486→          if (masked.openai?.apiKey) masked.openai.apiKey = masked.openai.apiKey.slice(0, 8) + '...';
   487→          if (masked.gemini?.apiKey) masked.gemini.apiKey = masked.gemini.apiKey.slice(0, 8) + '...';
   488→          if (masked.claude?.apiKey) masked.claude.apiKey = masked.claude.apiKey.slice(0, 8) + '...';
   489→          if (masked.custom?.apiKey) masked.custom.apiKey = masked.custom.apiKey.slice(0, 8) + '...';
   490→          res.writeHead(200, { 'Content-Type': 'application/json' });
   491→          res.end(JSON.stringify({ config: masked }));
   492→          return;
   493→        }
   494→
   495→        if (url === '/api/config/ai' && method === 'PUT') {
   496→          if (!body) {
   497→            res.writeHead(400, { 'Content-Type': 'application/json' });
   498→            res.end(JSON.stringify({ error: 'Request body is required' }));
   499→            return;
   500→          }
   501→          this.aiHandler.updateConfig(body);
   502→          res.writeHead(200, { 'Content-Type': 'application/json' });
   503→          res.end(JSON.stringify({ success: true }));
   504→          return;
   505→        }
   506→
   507→        // Scheduled messages
   508→        if (url === '/api/scheduler/messages' && method === 'GET') {
   509→          const sessionId = new URL(url, 'http://localhost').searchParams.get('sessionId');
   510→          const messages = this.scheduler.listPendingMessages(sessionId ?? undefined);
   511→          res.writeHead(200, { 'Content-Type': 'application/json' });
   512→          res.end(JSON.stringify({ messages }));
   513→          return;
   514→        }
   515→
   516→        if (url === '/api/scheduler/schedule' && method === 'POST') {
   517→          const { sessionId, jid, content, sendAt, recurring, type, broadcastJids } = body ?? {};
   518→          if (!sessionId || !jid || !content || !sendAt) {
   519→            res.writeHead(400, { 'Content-Type': 'application/json' });
   520→            res.end(JSON.stringify({ error: 'sessionId, jid, content, and sendAt are required' }));
   521→            return;
   522→          }
   523→          const msg = this.scheduler.scheduleMessage(sessionId, jid, content, sendAt, { recurring, type, broadcastJids });
   524→          res.writeHead(200, { 'Content-Type': 'application/json' });
   525→          res.end(JSON.stringify({ success: true, message: msg }));
   526→          return;
   527→        }
   528→
   529→        if (url === '/api/scheduler/cancel' && method === 'POST') {
   530→          const { messageId } = body ?? {};
   531→          if (!messageId) {
   532→            res.writeHead(400, { 'Content-Type': 'application/json' });
   533→            res.end(JSON.stringify({ error: 'messageId is required' }));
   534→            return;
   535→          }
   536→          const cancelled = this.scheduler.cancelScheduledMessage(messageId);
   537→          res.writeHead(200, { 'Content-Type': 'application/json' });
   538→          res.end(JSON.stringify({ success: cancelled }));
   539→          return;
   540→        }
   541→
   542→        // Send message
   543→        if (url === '/api/send' && method === 'POST') {
   544→          const { sessionId, jid, content } = body ?? {};
   545→          if (!sessionId || !jid || !content) {
   546→            res.writeHead(400, { 'Content-Type': 'application/json' });
   547→            res.end(JSON.stringify({ error: 'sessionId, jid, and content are required' }));
   548→            return;
   549→          }
   550→          try {
   551→            const result = await this.connectionManager.sendMessage(sessionId, jid, content);
   552→            res.writeHead(200, { 'Content-Type': 'application/json' });
   553→            res.end(JSON.stringify({ success: true, result }));
   554→          } catch (err) {
   555→            res.writeHead(500, { 'Content-Type': 'application/json' });
   556→            res.end(JSON.stringify({ error: String(err) }));
   557→          }
   558→          return;
   559→        }
   560→
   561→        // 404
   562→        res.writeHead(404, { 'Content-Type': 'application/json' });
   563→        res.end(JSON.stringify({ error: 'Not found', path: url }));
   564→      } catch (err) {
   565→        logger.error({ err, url, method }, 'HTTP request handler error');
   566→        res.writeHead(500, { 'Content-Type': 'application/json' });
   567→        res.end(JSON.stringify({ error: 'Internal server error' }));
   568→      }
   569→    });
   570→
   571→    return server;
   572→  }

   574→  private async connect(sessionId: string): Promise<void> {
   575→    try {
   576→      await this.sessionManager.createSession(sessionId);
   577→
   578→      const sock = await this.connectionManager.createConnection({
   579→        sessionId,
   580→        printQR: true,
   581→        browser: 'CALTEX MD',
   582→        autoReconnect: true,
   583→        maxReconnectAttempts: 10,
   584→        reconnectBaseDelay: 2000,
   585→      });
   586→
   587→      this.scheduler.registerSocket(sessionId, sock);
   588→
   589→      logger.info({ sessionId }, 'Bot session connecting...');
   590→    } catch (err) {
   591→      logger.error({ err, sessionId }, 'Failed to create connection');
   592→      throw err;
   593→    }
   594→  }

   596→  private setupShutdownHandlers(): void {
   597→    const shutdown = async (signal: string) => {
   598→      logger.info({ signal }, 'Shutdown signal received, cleaning up...');
   599→
   600→      try {
   601→        await this.connectionManager.disconnectAll();
   602→        this.scheduler.destroy();
   603→        this.apiClient.destroy();
   604→        this.aiHandler.destroy();
   605→
   606→        this.httpServer.close(() => {
   607→          logger.info('HTTP server closed');
   608→          process.exit(0);
   609→        });
   610→
   611→        setTimeout(() => {
   612→          logger.warn('Forced shutdown after timeout');
   613→          process.exit(1);
   614→        }, 10000);
   615→      } catch (err) {
   616→        logger.error({ err }, 'Error during shutdown');
   617→        process.exit(1);
   618→      }
   619→    };
   620→
   621→    process.on('SIGINT', () => shutdown('SIGINT'));
   622→    process.on('SIGTERM', () => shutdown('SIGTERM'));
   623→    process.on('SIGQUIT', () => shutdown('SIGQUIT'));
   624→  }

   626→  private formatUptime(ms: number): string {
   627→    const seconds = Math.floor(ms / 1000);
   628→    const minutes = Math.floor(seconds / 60);
   629→    const hours = Math.floor(minutes / 60);
   630→    const days = Math.floor(hours / 24);
   631→
   632→    if (days > 0) return `${days}d ${hours % 24}h ${minutes % 60}m`;
   633→    if (hours > 0) return `${hours}h ${minutes % 60}m ${seconds % 60}s`;
   634→    if (minutes > 0) return `${minutes}m ${seconds % 60}s`;
   635→    return `${seconds}s`;
   636→  }
